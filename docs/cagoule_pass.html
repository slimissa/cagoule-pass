129    @classmethod
130    def open(cls, password: bytes, vault_dir: Path = DEFAULT_VAULT_DIR) -> "Vault":
131        """
132        Ouvre et déchiffre un coffre existant.
133
134        Raises:
135            VaultNotFoundError : si le coffre n'existe pas
136            VaultAuthError     : si le mot de passe est incorrect
137            VaultCorruptError  : si le fichier est corrompu
138        """
139        vault_path = vault_dir / VAULT_FILE
140        if not vault_path.exists():
141            raise VaultNotFoundError(
142                f"Aucun coffre trouvé dans {vault_dir}. "
143                "Initialisez-en un avec : cagoule-pass init"
144            )
145
146        raw = vault_path.read_bytes()
147
148        try:
149            from cagoule import decrypt, CagouleAuthError
150            plaintext = decrypt(raw, password)
151        except Exception as e:
152            if "auth" in str(e).lower() or "InvalidTag" in str(type(e)):
153                raise VaultAuthError("Mot de passe maître incorrect.") from e
154            # Essayer de distinguer les erreurs d'auth des erreurs de format
155            err_name = type(e).__name__
156            if "Auth" in err_name:
157                raise VaultAuthError("Mot de passe maître incorrect.") from e
158            raise VaultCorruptError(f"Coffre illisible : {e}") from e
159
160        try:
161            data = json.loads(plaintext.decode("utf-8"))
162            entries = [Entry.from_dict(e) for e in data.get("entries", [])]
163            created = data.get("created", _now())
164        except (json.JSONDecodeError, KeyError, TypeError) as e:
165            raise VaultCorruptError(f"Structure du coffre invalide : {e}") from e
166
167        return cls(entries=entries, created=created, vault_dir=vault_dir)

169    def save(self, password: bytes) -> None:
170        """
171        Chiffre et sauvegarde le coffre sur disque.
172
173        Le fichier est écrit atomiquement (fichier temporaire + rename).
174        """
175        data = {
176            "version": VAULT_SCHEMA_VER,
177            "created": self._created,
178            "entries": [e.to_dict() for e in self._entries],

304    def import_json(self, path: Path, overwrite: bool = False) -> tuple[int, int]:
305        """
306        Importe des entrées depuis un JSON exporté.
307
308        Returns:
309            (importées, ignorées) : tuple de compteurs
310        """
311        data = json.loads(path.read_text())
312        imported = skipped = 0
313
314        for e_dict in data.get("entries", []):
315            entry = Entry.from_dict(e_dict)
316            existing = self._find(entry.service)
317            if existing is not None:
318                if overwrite:
319                    self._entries.remove(existing)
320                    self._entries.append(entry)
321                    imported += 1
322                else:
323                    skipped += 1
324            else:
325                self._entries.append(entry)
326                imported += 1
327
328        return imported, skipped

def generate( length: int = 16, use_upper: bool = True, use_digits: bool = True, use_symbols: bool = False, no_ambiguous: bool = False) -> str: 
 38def generate(
 39    length:       int  = 16,
 40    use_upper:    bool = True,
 41    use_digits:   bool = True,
 42    use_symbols:  bool = False,
 43    no_ambiguous: bool = False,
 44) -> str:
 45    """
 46    Génère un mot de passe aléatoire sécurisé.
 47
 48    Args:
 49        length       : longueur du mot de passe (min 4)
 50        use_upper    : inclure les majuscules
 51        use_digits   : inclure les chiffres
 52        use_symbols  : inclure les symboles spéciaux
 53        no_ambiguous : exclure les caractères ambigus (0, O, l, I, 1...)
 54
 55    Returns:
 56        str : mot de passe généré
 57
 58    Raises:
 59        ValueError : si length < 4 ou alphabet vide
 60    """
 61    if length < 4:
 62        raise ValueError(f"Longueur minimale : 4 caractères (reçu {length})")
 63
 64    # Construire l'alphabet
 65    alphabet = LOWER
 66    required: list[str] = []
 67
 68    if use_upper:
 69        alphabet += UPPER
 70        required.append(UPPER)
 71
 72    if use_digits:
 73        alphabet += DIGITS
 74        required.append(DIGITS)
 75
 76    if use_symbols:
 77        alphabet += SYMBOLS
 78        required.append(SYMBOLS)
 79
 80    # Exclure les caractères ambigus si demandé
 81    if no_ambiguous:
 82        alphabet = "".join(c for c in alphabet if c not in AMBIGUOUS)
 83        required = [
 84            "".join(c for c in cat if c not in AMBIGUOUS)
 85            for cat in required
 86        ]
 87        required = [cat for cat in required if cat]  # retirer les vides
 88
 89    if not alphabet:
 90        raise ValueError("L'alphabet est vide après filtrage.")
 91
 92    # S'assurer qu'il y a assez de place pour les caractères requis
 93    if len(required) >= length:
 94        raise ValueError(
 95            f"Longueur ({length}) insuffisante pour inclure tous les types de caractères requis."
 96        )
 97
 98    # Générer le mot de passe
 99    # 1. Un caractère obligatoire de chaque catégorie requise
100    forced = [_secure_choice(cat) for cat in required]
101
102    # 2. Remplir le reste
103    rest = [_secure_choice(alphabet) for _ in range(length - len(forced))]
104
105    # 3. Mélanger (Fisher-Yates via os.urandom)
106    password = forced + rest
107    for i in range(len(password) - 1, 0, -1):
108        j = int.from_bytes(os.urandom(2), "big") % (i + 1)
109        password[i], password[j] = password[j], password[i]
110
111    return "".join(password)

def entropy_bits(password: str) -> float: 
114def entropy_bits(password: str) -> float:
115    """
116    Estime l'entropie du mot de passe en bits.
117
118    Formule : log2(|alphabet|^longueur) = longueur × log2(|alphabet|)
119    Note : estimation basée sur la taille de l'alphabet détecté.
120    """
121    has_lower   = any(c in LOWER   for c in password)
122    has_upper   = any(c in UPPER   for c in password)
123    has_digit   = any(c in DIGITS  for c in password)
124    has_symbol  = any(c in SYMBOLS for c in password)
125
126    pool = 0
127    if has_lower:  pool += len(LOWER)
128    if has_upper:  pool += len(UPPER)
129    if has_digit:  pool += len(DIGITS)
130    if has_symbol: pool += len(SYMBOLS)
131
132    if pool == 0:
133        return 0.0
134
135    return round(len(password) * math.log2(pool), 1)
